Redux Saga Helpers

  • takeEvery — spawns saga on each dispatched action
  • takeLatest — spawns saga on the last dispatched action
  • and other 
          import { call, put, takeEvery } from 'redux-saga/effects';
           
          function* getUser(action) {
            try {
              const user = yield call(userApi.fetchUser, action.payload.userId);
              yield put({ type: UserActionType.GET_USER_SUCCESS, user: user });
            } catch (err) {
              yield put({ type: UserActionType.GET_USER_FAIL, message: err.message });
            }
          }
           
          function* userSaga() {
            yield takeEvery(UserActionType.GET_USER, getUser);
          }

        import { createRoutine } from 'redux-saga-routines';
         
        const routine = createRoutine('ACTION_TYPE');
         
        // Action Types. Also: TRIGGER, REQUEST, FULFILL
        console.log(routine.SUCCESS); // 'ACTION_TYPE/SUCCESS';
        console.log(routine.FAILURE); // 'ACTION_TYPE/FAILURE';
         
        // Action Creators. Also: trigger, request, fulfill
        console.log(routine.success(payload)); // { type: 'ACTION_TYPE/SUCCESS', payload };
        console.log(routine.failure(payload)); // { type: 'ACTION_TYPE/FAILURE', payload };
